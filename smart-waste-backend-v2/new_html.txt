<script>
const video = document.getElementById('video');
const canvas = document.getElementById('canvas');
const history = [];
const MAX_HISTORY = 5;

// Start camera
async function startCamera() {
    try {
        const stream = await navigator.mediaDevices.getUserMedia({
            video: { facingMode: 'environment', width: { ideal: 640 }, height: { ideal: 480 } }
        });
        video.srcObject = stream;
    } catch(e) {
        console.error('Camera error:', e);
        alert('Camera access denied. Please allow camera permission.');
    }
}

// Capture and analyze
async function capture() {
    const btn = document.getElementById('captureBtn');
    const loader = document.getElementById('loader');
    
    btn.disabled = true;
    loader.classList.add('visible');
    
    // Capture frame
    canvas.width = video.videoWidth || 640;
    canvas.height = video.videoHeight || 480;
    canvas.getContext('2d').drawImage(video, 0, 0);
    
    canvas.toBlob(async (blob) => {
        const formData = new FormData();
        formData.append('file', blob, 'capture.jpg');
        
        const binId = document.getElementById('binSelect').value;
        
        try {
            // Call dual analysis endpoint
            const response = await fetch(
                `/analyze/dual-analysis?bin_id=${binId}`,
                { method: 'POST', body: formData }
            );
            const data = await response.json();
            
            if (data.success) {
                showResults(data);
                addToHistory(data);
                // Also update IoT logic to map to dashboard
                await syncToDashboard(binId, data);
            }
        } catch(e) {
            console.error('Analysis error:', e);
            alert('Analysis failed. Check backend connection.');
        } finally {
            btn.disabled = false;
            loader.classList.remove('visible');
        }
    }, 'image/jpeg', 0.85);
}

// Map the dual response back to the original IoT update so the map still reflects it
async function syncToDashboard(binId, data) {
    try {
        await fetch(`${window.location.origin}/iot/update`, {
            method: 'POST',
            headers: {
                'Content-Type': 'application/json',
                'X-API-Key': 'smartbin-iot-key-2026'
            },
            body: JSON.stringify({
                dustbin_id: parseInt(binId),
                fill_level: data.fill_level?.fill_level || 50,
                status:     data.fill_level?.status || 'half-full',
                distance:   50 - ((data.fill_level?.fill_level || 50) * 0.5),
                battery:    85,
                source:     'phone_camera',
                timestamp:  new Date().toISOString()
            })
        });
    } catch(e) {
        console.error('IoT sync error:', e);
    }
}

// Fill level colors
const fillColors = {
    'empty': '#22c55e',
    'half-full': '#eab308', 
    'full': '#f97316',
    'overflowing': '#ef4444'
};

// Waste type colors
const wasteColors = {
    'organic': '#22c55e',
    'recyclable': '#3b82f6',
    'hazardous': '#ef4444',
    'dry': '#eab308'
};

const wasteIcons = {
    'organic': '🌿',
    'recyclable': '♻️',
    'hazardous': '⚠️',
    'dry': '🧺'
};

function showResults(data) {
    const fillData = data.fill_level;
    const wasteData = data.waste_type;
    
    // Fill level card
    const fillCard = document.getElementById('fillCard');
    fillCard.className = 'result-card visible ' + (fillData?.status || '');
    document.getElementById('fillValue').textContent = 
        (fillData?.status || 'Unknown').toUpperCase();
    document.getElementById('fillSub').textContent = 
        `${fillData?.fill_level || 0}% full`;
    const fillColor = fillColors[fillData?.status] || '#94a3b8';
    document.getElementById('fillBar').style.width = 
        (fillData?.fill_level || 0) + '%';
    document.getElementById('fillBar').style.background = fillColor;
    
    // Waste type card
    const wasteCard = document.getElementById('wasteCard');
    const category = wasteData?.category || 'unknown';
    wasteCard.className = 'result-card visible ' + category;
    const wasteIcon = wasteIcons[category] || '🗑️';
    document.getElementById('wasteValue').textContent = 
        wasteIcon + ' ' + category.charAt(0).toUpperCase() + category.slice(1);
    document.getElementById('wasteSub').textContent = 
        `${wasteData?.confidence || 0}% confidence`;
    const wasteColor = wasteColors[category] || '#94a3b8';
    document.getElementById('wasteBar').style.width = 
        (wasteData?.confidence || 0) + '%';
    document.getElementById('wasteBar').style.background = wasteColor;
    
    // Eco tip
    if (wasteData?.circular_economy?.action) {
        document.getElementById('ecoTipText').textContent = 
            wasteData.circular_economy.action + ' — ' + 
            (wasteData.disposal_tip || '');
        document.getElementById('ecoTip').classList.add('visible');
    }
    
    // Overflow alert + vibration
    const overflowAlert = document.getElementById('overflowAlert');
    if (fillData?.status === 'overflowing') {
        overflowAlert.classList.add('visible');
        if (navigator.vibrate) navigator.vibrate([200, 100, 200, 100, 400]);
    } else {
        overflowAlert.classList.remove('visible');
    }
}

function addToHistory(data) {
    const item = {
        fill: data.fill_level?.status || 'unknown',
        waste: data.waste_type?.category || 'unknown',
        confidence: data.waste_type?.confidence || 0,
        time: new Date().toLocaleTimeString()
    };
    history.unshift(item);
    if (history.length > MAX_HISTORY) history.pop();
    
    const histDiv = document.getElementById('history');
    histDiv.innerHTML = history.map(h => `
        <div class="history-item">
            <span>
                ${wasteIcons[h.waste] || '🗑️'} ${h.waste} 
                · ${h.fill}
            </span>
            <span style="color:#64748b">${h.time}</span>
        </div>
    `).join('');
}

// Start camera on load
startCamera();

// Auto capture every 5 seconds
setInterval(() => {
    if (!document.getElementById('captureBtn').disabled) {
        capture();
    }
}, 5000);
</script>
